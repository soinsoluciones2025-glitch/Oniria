@Composable
fun SettingsScreen(
    viewModel: IntentionViewModel = hiltViewModel()
) {
    val uiState by viewModel.uiState.collectAsStateWithLifecycle()

    Column(
        modifier = Modifier
            .fillMaxSize()
            .padding(16.dp),
        horizontalAlignment = Alignment.CenterHorizontally
    ) {
        Text(
            text = "Configuración",
            style = MaterialTheme.typography.headlineMedium
        )
        Text(
            text = "Aquí podrás ajustar las preferencias de la aplicación.",
            textAlign = TextAlign.Center,
            modifier = Modifier.padding(top = 8.dp)
        )

        Spacer(modifier = Modifier.height(24.dp))
        Divider()

        HearingModeSetting(
            isChecked = uiState.isHearingModeEnabled,
            onCheckedChange = { viewModel.onHearingModeChanged(it) }
        )

        Divider(modifier = Modifier.padding(vertical = 8.dp))

        VoiceSelectionSetting(
            availableVoices = uiState.availableVoices,
            selectedVoiceId = uiState.selectedVoiceId,
            onVoiceSelected = { viewModel.onVoiceSelected(it) }
        )
    }
}

@Composable
fun HearingModeSetting(
    isChecked: Boolean,
    onCheckedChange: (Boolean) -> Unit,
    modifier: Modifier = Modifier
) {
    Row(
        modifier = modifier
            .fillMaxWidth()
            .padding(vertical = 16.dp),
        verticalAlignment = Alignment.CenterVertically,
        horizontalArrangement = Arrangement.SpaceBetween
    ) {
        Column(modifier = Modifier.weight(1f)) {
            Text("Modo de Accesibilidad Auditiva", fontSize = 18.sp, style = MaterialTheme.typography.bodyLarge)
            Text(
                "Traduce la voz del entorno a texto en tiempo real.",
                fontSize = 14.sp,
                style = MaterialTheme.typography.bodySmall
            )
        }
        Switch(
            checked = isChecked,
            onCheckedChange = onCheckedChange
        )
    }
}

@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun VoiceSelectionSetting(
    availableVoices: List<String>,
    selectedVoiceId: String?,
    onVoiceSelected: (String) -> Unit,
    modifier: Modifier = Modifier
) {
    var expanded by remember { mutableStateOf(false) }
    
    Row(
        modifier = modifier
            .fillMaxWidth()
            .padding(vertical = 16.dp),
        verticalAlignment = Alignment.CenterVertically,
        horizontalArrangement = Arrangement.SpaceBetween
    ) {
        Column(modifier = Modifier.weight(1f)) {
            Text("Voz del Sintetizador", fontSize = 18.sp, style = MaterialTheme.typography.bodyLarge)
            Text(
                "Elige la voz que usará la aplicación para hablar.",
                fontSize = 14.sp,
                style = MaterialTheme.typography.bodySmall
            )
            Spacer(modifier = Modifier.height(8.dp))

            if (availableVoices.isNotEmpty()) {
                ExposedDropdownMenuBox(
                    expanded = expanded,
                    onExpandedChange = { expanded = !expanded }
                ) {
                    OutlinedTextField(
                        value = selectedVoiceId ?: "Voz por defecto",
                        onValueChange = {},
                        readOnly = true,
                        trailingIcon = {
                            ExposedDropdownMenuDefaults.TrailingIcon(expanded = expanded)
                        },
                        modifier = Modifier
                            .menuAnchor()
                            .fillMaxWidth()
                    )
                    ExposedDropdownMenu(
                        expanded = expanded,
                        onDismissRequest = { expanded = false }
                    ) {
                        availableVoices.forEach { voiceId ->
                            DropdownMenuItem(
                                text = { Text(voiceId) },
                                onClick = {
                                    onVoiceSelected(voiceId)
                                    expanded = false
                                }
                            )
                        }
                    }
                }
            } else {
                 Text(
                    "No hay voces adicionales disponibles o el motor de voz no se ha cargado.",
                    fontSize = 14.sp,
                    style = MaterialTheme.typography.bodySmall
                )
            }
        }
    }
}
