// Purpose: The primary user interface for communication. This Jetpack Compose screen provides
// a text area for composing messages, large buttons for core actions (speak, clear, AI assist),
// a grid of quick-access phrases, and the new Auditory Accessibility Mode.
package com.oniria.ui.screens

import androidx.compose.foundation.background
import androidx.compose.foundation.layout.*
import androidx.compose.foundation.lazy.grid.GridCells
import androidx.compose.foundation.lazy.grid.LazyVerticalGrid
import androidx.compose.foundation.lazy.grid.items
import androidx.compose.material3.*
import androidx.compose.runtime.Composable
import androidx.compose.runtime.getValue
import androidx.compose.ui.Alignment
import androidx.compose.ui.Modifier
import androidx.compose.ui.graphics.Color
import androidx.compose.ui.text.TextStyle
import androidx.compose.ui.text.font.FontWeight
import androidx.compose.ui.text.style.TextAlign
import androidx.compose.ui.unit.dp
import androidx.compose.ui.unit.sp
import androidx.hilt.navigation.compose.hiltViewModel
import androidx.lifecycle.compose.collectAsStateWithLifecycle
import com.oniria.viewmodels.IntentionViewModel

val commonPhrases = listOf(
    "Hola", "Adiós", "Gracias", "Por favor",
    "Tengo hambre", "Tengo sed", "Necesito ayuda", "¿Puedes venir?"
)

@Composable
fun CommunicationScreen(
    viewModel: IntentionViewModel = hiltViewModel()
) {
    val uiState by viewModel.uiState.collectAsStateWithLifecycle()

    if (uiState.isHearingModeEnabled) {
        HearingModeUI(transcribedText = uiState.transcribedText)
    } else {
        StandardModeUI(viewModel = viewModel)
    }
}

@Composable
fun StandardModeUI(viewModel: IntentionViewModel) {
    val uiState by viewModel.uiState.collectAsStateWithLifecycle()

    Column(
        modifier = Modifier
            .fillMaxSize()
            .padding(16.dp),
        verticalArrangement = Arrangement.spacedBy(16.dp)
    ) {
        // Main text display and input area
        OutlinedTextField(
            value = uiState.currentText,
            onValueChange = { viewModel.onTextChanged(it) },
            modifier = Modifier
                .fillMaxWidth()
                .weight(1f),
            textStyle = TextStyle(fontSize = 28.sp),
            label = { Text("Escribe o selecciona una frase") }
        )

        // Action Buttons Row
        Column(
            modifier = Modifier.fillMaxWidth(),
            verticalArrangement = Arrangement.spacedBy(8.dp)
        ) {
            Row(
                modifier = Modifier.fillMaxWidth(),
                horizontalArrangement = Arrangement.spacedBy(8.dp, Alignment.CenterHorizontally)
            ) {
                Button(
                    onClick = { viewModel.onSpeakClicked() },
                    enabled = !uiState.isSpeaking && uiState.currentText.isNotBlank(),
                    modifier = Modifier.weight(1f)
                ) {
                    Text(if (uiState.isSpeaking) "Hablando..." else "Hablar", fontSize = 18.sp)
                }
                Button(
                    onClick = { viewModel.onClearClicked() },
                    modifier = Modifier.weight(1f)
                ) {
                    Text("Limpiar", fontSize = 18.sp)
                }
            }
            Row(
                modifier = Modifier.fillMaxWidth(),
                horizontalArrangement = Arrangement.spacedBy(8.dp, Alignment.CenterHorizontally)
            ) {
                Button(
                    onClick = { viewModel.onCorrectTextClicked() },
                    enabled = !uiState.isProcessing && uiState.currentText.isNotBlank(),
                    modifier = Modifier.weight(1f)
                ) {
                    Text("Corregir (IA)", fontSize = 18.sp)
                }
                Button(
                    onClick = { viewModel.onRephraseClicked() },
                    enabled = !uiState.isProcessing && uiState.currentText.isNotBlank(),
                    modifier = Modifier.weight(1f)
                ) {
                     Text("Reformular (IA)", fontSize = 18.sp)
                }
            }
             if (uiState.isProcessing) {
                LinearProgressIndicator(modifier = Modifier.fillMaxWidth().padding(top=8.dp))
            }
        }
        
        Divider()

        // Grid of common phrases
        LazyVerticalGrid(
            columns = GridCells.Adaptive(minSize = 150.dp),
            modifier = Modifier.weight(1f),
            verticalArrangement = Arrangement.spacedBy(8.dp),
            horizontalArrangement = Arrangement.spacedBy(8.dp)
        ) {
            items(commonPhrases) { phrase ->
                Button(
                    onClick = { viewModel.onPhraseSelected(phrase) },
                    modifier = Modifier.fillMaxWidth().height(80.dp)
                ) {
                    Text(phrase, fontSize = 18.sp, textAlign = TextAlign.Center)
                }
            }
        }
    }
}

@Composable
fun HearingModeUI(transcribedText: String) {
    Box(
        modifier = Modifier
            .fillMaxSize()
            .background(Color.Black)
            .padding(24.dp),
        contentAlignment = Alignment.Center
    ) {
        Text(
            text = transcribedText,
            color = Color.White,
            fontSize = 48.sp,
            fontWeight = FontWeight.Bold,
            textAlign = TextAlign.Center,
            lineHeight = 60.sp
        )
    }
}
