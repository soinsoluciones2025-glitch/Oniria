// a text area for composing messages, large buttons for core actions (speak, clear, AI assist),
// a grid of quick-access phrases, and the new Auditory Accessibility Mode.

val commonPhrases = listOf(
    "Hola", "Adiós", "Gracias", "Por favor",
    "Tengo hambre", "Tengo sed", "Necesito ayuda", "¿Puedes venir?",
    "Sí", "No", "No lo sé", "Espera un momento"
)

@OptIn(ExperimentalPermissionsApi::class)
@Composable
fun CommunicationScreen(
    viewModel: IntentionViewModel = hiltViewModel()
) {
    val uiState by viewModel.uiState.collectAsStateWithLifecycle()
    val audioPermissionState = rememberPermissionState(Manifest.permission.RECORD_AUDIO)


    if (uiState.isHearingModeEnabled) {
        if (audioPermissionState.status.isGranted) {
            HearingModeUI(transcribedText = uiState.transcribedText)
        } else {
            RequestAudioPermissionUI {
                audioPermissionState.launchPermissionRequest()
            }
        }
    } else {
        StandardModeUI(viewModel = viewModel)
    }
}

@Composable
fun StandardModeUI(viewModel: IntentionViewModel) {
    val uiState by viewModel.uiState.collectAsStateWithLifecycle()

    Column(
        modifier = Modifier
            .fillMaxSize()
            .padding(horizontal = 16.dp, vertical = 12.dp),
        verticalArrangement = Arrangement.spacedBy(12.dp)
    ) {
        // Main text display and input area
        OutlinedTextField(
            value = uiState.currentText,
            onValueChange = { viewModel.onTextChanged(it) },
            modifier = Modifier
                .fillMaxWidth()
                .weight(1f),
            textStyle = TextStyle(fontSize = 24.sp),
            placeholder = { Text("Escribe o selecciona una frase...") },
            isError = uiState.errorMessage != null
        )

        AnimatedVisibility(visible = uiState.errorMessage != null) {
            Text(
                text = uiState.errorMessage ?: "",
                color = MaterialTheme.colorScheme.error,
                modifier = Modifier.padding(horizontal = 8.dp)
            )
        }

        // Action Buttons Grid
        Column(
            modifier = Modifier.fillMaxWidth(),
            verticalArrangement = Arrangement.spacedBy(8.dp)
        ) {
            Row(
                modifier = Modifier.fillMaxWidth(),
                horizontalArrangement = Arrangement.spacedBy(8.dp)
            ) {
                ActionButton(
                    text = if (uiState.isSpeaking) "Hablando..." else "Hablar",
                    icon = Icons.Default.VolumeUp,
                    onClick = { viewModel.onSpeakClicked() },
                    enabled = !uiState.isSpeaking && uiState.currentText.isNotBlank(),
                    modifier = Modifier.weight(1f),
                    color = MaterialTheme.colorScheme.primary
                )
                ActionButton(
                    text = "Limpiar",
                    icon = Icons.Default.Clear,
                    onClick = { viewModel.onClearClicked() },
                    modifier = Modifier.weight(1f),
                    color = MaterialTheme.colorScheme.secondary
                )
            }
            Row(
                modifier = Modifier.fillMaxWidth(),
                horizontalArrangement = Arrangement.spacedBy(8.dp)
            ) {
                ActionButton(
                    text = "Corregir (IA)",
                    icon = Icons.Default.Spellcheck,
                    onClick = { viewModel.onCorrectTextClicked() },
                    enabled = !uiState.isProcessing && uiState.currentText.isNotBlank(),
                    modifier = Modifier.weight(1f),
                    color = MaterialTheme.colorScheme.tertiary
                )
                ActionButton(
                    text = "Reformular (IA)",
                    icon = Icons.Default.AutoAwesome,
                    onClick = { viewModel.onRephraseClicked() },
                    enabled = !uiState.isProcessing && uiState.currentText.isNotBlank(),
                    modifier = Modifier.weight(1f),
                    color = MaterialTheme.colorScheme.tertiary
                )
            }
             if (uiState.isProcessing) {
                LinearProgressIndicator(modifier = Modifier
                    .fillMaxWidth()
                    .padding(top = 8.dp))
            }
        }

        // Grid of common phrases
        LazyVerticalGrid(
            columns = GridCells.Adaptive(minSize = 140.dp),
            modifier = Modifier.weight(0.8f),
            verticalArrangement = Arrangement.spacedBy(8.dp),
            horizontalArrangement = Arrangement.spacedBy(8.dp),
            contentPadding = PaddingValues(top = 8.dp)
        ) {
            items(commonPhrases) { phrase ->
                FilledTonalButton(
                    onClick = { viewModel.onPhraseSelected(phrase) },
                    modifier = Modifier.height(60.dp)
                ) {
                    Text(phrase, textAlign = TextAlign.Center)
                }
            }
        }
    }
}

@Composable
fun ActionButton(
    text: String,
    icon: androidx.compose.ui.graphics.vector.ImageVector,
    onClick: () -> Unit,
    modifier: Modifier = Modifier,
    enabled: Boolean = true,
    color: Color
) {
    Button(
        onClick = onClick,
        enabled = enabled,
        modifier = modifier.height(56.dp),
        colors = ButtonDefaults.buttonColors(containerColor = color)
    ) {
        Icon(icon, contentDescription = null, modifier = Modifier.size(20.dp))
        Spacer(Modifier.width(8.dp))
        Text(text, fontSize = 16.sp)
    }
}


@Composable
fun HearingModeUI(transcribedText: String) {
    Box(
        modifier = Modifier
            .fillMaxSize()
            .background(MaterialTheme.colorScheme.surfaceVariant)
            .padding(24.dp),
        contentAlignment = Alignment.Center
    ) {
        Text(
            text = transcribedText,
            color = MaterialTheme.colorScheme.onSurfaceVariant,
            fontSize = 42.sp,
            fontWeight = FontWeight.Bold,
            textAlign = TextAlign.Center,
            lineHeight = 54.sp
        )
    }
}

@Composable
fun RequestAudioPermissionUI(onRequestPermission: () -> Unit) {
    Column(
        modifier = Modifier
            .fillMaxSize()
            .padding(24.dp),
        horizontalAlignment = Alignment.CenterHorizontally,
        verticalArrangement = Arrangement.Center
    ) {
        Text(
            "Permiso de Micrófono Requerido",
            style = MaterialTheme.typography.headlineSmall,
            textAlign = TextAlign.Center
        )
        Spacer(modifier = Modifier.height(16.dp))
        Text(
            "El modo de accesibilidad auditiva necesita acceso a tu micrófono para transcribir el sonido ambiente a texto.",
            textAlign = TextAlign.Center,
            style = MaterialTheme.typography.bodyLarge
        )
        Spacer(modifier = Modifier.height(24.dp))
        Button(onClick = onRequestPermission) {
            Text("Otorgar Permiso")
        }
    }
}
